<script>

// AUTO TARIKH + HARI
const hari = ["AHAD","ISNIN","SELASA","RABU","KHAMIS","JUMAAT","SABTU"];
let now = new Date();
let tarikh = now.toLocaleDateString('en-GB');
let masa = now.toTimeString().slice(0,5);
let hariIni = hari[now.getDay()];

document.getElementById("tarikh").value = `${tarikh}@${masa}H / ${hariIni}`;

// DATA KINTA (BASIC - BOLEH UPGRADE LAGI)
const data = {
  "P063 Tambun": {
    "Hulu Kinta": ["Ulu Kinta", "Tanjung Rambutan"],
    "Manjoi": ["Manjoi"]
  },
  "P068 Gopeng": {
    "Teja": ["Teja", "Kota Baharu"],
    "Sungai Raya": ["Sungai Raya"]
  }
};

function updateDun(){
  let p = document.getElementById("parlimen").value;
  let dun = document.getElementById("dun");
  dun.innerHTML = "<option>Pilih DUN</option>";

  for(let d in data[p]){
    dun.innerHTML += `<option>${d}</option>`;
  }
}

function updateMukim(){
  let p = document.getElementById("parlimen").value;
  let d = document.getElementById("dun").value;
  let mukim = document.getElementById("mukim");
  mukim.innerHTML = "<option>Pilih Mukim</option>";

  data[p][d].forEach(m=>{
    mukim.innerHTML += `<option>${m}</option>`;
  });
}

function val(id){ return document.getElementById(id).value; }

// GENERATE REPORT
function generate(){

let peg = val("pegawai") || 0;
let ang = val("anggota") || 0;

let report = `Assalamualaikum dan Salam Sejahtera YB Dato’ Sri / Dato’ / Datuk / Tuan / Puan,

*LAPORAN PEMANTAUAN KAWASAN BERISIKO BANJIR DAERAH KINTA*

${val("tarikh")}

Parlimen: ${val("parlimen")}
Dun: ${val("dun")}
Mukim: ${val("mukim")}


2. Kawasan/Lokasi Pemantauan:
a. ${val("lokasi1")}
b. ${val("lokasi2")}
c. ${val("lokasi3")}


3. Jalan Yang Dinaiki Air / Terputus: ${val("jalan")}


4. Kekuatan Anggota & Logistik:

APM:
Pegawai : ${peg} x Pegawai
Anggota : ${ang} x Anggota LLP
Kenderaan : ${val("kenderaan")}


Catatan:
1. Kawasan dinaiki air: ${val("air")}
2. Aras air: ${val("trend")}
3. Cuaca: ${val("cuaca")}


Sekian untuk makluman.


Disediakan Oleh:
Pusat Kawalan Operasi Daerah (PKOD)
Angkatan Pertahanan Awam Malaysia
Daerah Kinta`;

document.getElementById("output").innerText = report;
}

// COPY
function copy(){
navigator.clipboard.writeText(document.getElementById("output").innerText);
alert("Copied!");
}

// WHATSAPP
function whatsapp(){
let text = encodeURIComponent(document.getElementById("output").innerText);
window.open("https://wa.me/?text=" + text);
}
